# Snackbar Success Message
Display a Swoosh animation snackbar in order to represent a success status

### Usage

```typescript
import { Component } from '@angular/core';
import { MatXtndSnackbarSuccessService } from '@matxtnd'
...
@Component({ ... })
...
constructor(private readonly snackbarSuccessService: MatXtndSnackbarSuccessService) { }
...
this.snackbarSuccessService.Open('Some Success Message');
...  

```
